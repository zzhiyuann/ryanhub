import Foundation

// MARK: - CaffeineTrackerViewModel

@Observable
@MainActor
final class CaffeineTrackerViewModel {

    // MARK: - State

    var entries: [CaffeineTrackerEntry] = []
    var isLoading = false
    var errorMessage: String?
    var selectedTab: CaffeineTrackerTab = .today
    var dailyGoal: Int = 400
    var showingDrinkEntry = false

    // Updated every 60s to force decay recomputation
    private(set) var decayTick = Date()
    private var decayTimer: Timer?

    // MARK: - Tab

    enum CaffeineTrackerTab: String, CaseIterable, Identifiable {
        case today = "Today"
        case trends = "Trends"
        var id: String { rawValue }
    }

    // MARK: - Bridge URL

    private var bridgeBaseURL: String {
        UserDefaults.standard.string(forKey: "ryanhub_server_url")
            .flatMap { URL(string: $0)?.host }
            .map { "http://\($0):18790" }
            ?? "http://localhost:18790"
    }

    // MARK: - Init / Deinit

    init() {
        Task { await loadData() }
        startDecayTimer()
    }

    deinit {
        decayTimer?.invalidate()
    }

    private func startDecayTimer() {
        decayTimer = Timer.scheduledTimer(withTimeInterval: 60, repeats: true) { [weak self] _ in
            Task { @MainActor in
                self?.decayTick = Date()
            }
        }
    }

    // MARK: - Base Caffeine (mg) Per Drink Type

    static func baseMg(for drinkType: DrinkType) -> Int {
        switch drinkType {
        case .espresso: return 63
        case .dripCoffee: return 95
        case .latte: return 63
        case .coldBrew: return 200
        case .cappuccino: return 63
        case .americano: return 63
        case .greenTea: return 28
        case .blackTea: return 47
        case .energyDrink: return 80
        case .decaf: return 2
        case .custom: return 100
        }
    }

    static func sizeMultiplier(for size: DrinkSize) -> Double {
        switch size {
        case .small: return 1.0
        case .medium: return 1.5
        case .large: return 2.0
        }
    }

    static func defaultCaffeineMg(drinkType: DrinkType, size: DrinkSize) -> Int {
        Int(Double(baseMg(for: drinkType)) * sizeMultiplier(for: size))
    }

    // MARK: - CRUD

    func loadData() async {
        isLoading = true
        errorMessage = nil
        defer { isLoading = false }

        guard let url = URL(string: "\(bridgeBaseURL)/modules/caffeineTracker/data") else {
            errorMessage = "Invalid URL"
            return
        }

        do {
            let (data, _) = try await URLSession.shared.data(from: url)
            let decoder = JSONDecoder()
            decoder.dateDecodingStrategy = .iso8601
            entries = try decoder.decode([CaffeineTrackerEntry].self, from: data)
            cacheData()
        } catch {
            errorMessage = error.localizedDescription
        }
    }

    func addEntry(_ entry: CaffeineTrackerEntry) async {
        guard let url = URL(string: "\(bridgeBaseURL)/modules/caffeineTracker/data/add") else { return }

        do {
            var request = URLRequest(url: url)
            request.httpMethod = "POST"
            request.setValue("application/json", forHTTPHeaderField: "Content-Type")
            let encoder = JSONEncoder()
            encoder.dateEncodingStrategy = .iso8601
            request.httpBody = try encoder.encode(entry)
            let (_, _) = try await URLSession.shared.data(for: request)
            entries.append(entry)
            cacheData()
        } catch {
            errorMessage = error.localizedDescription
        }
    }

    func deleteEntry(_ entry: CaffeineTrackerEntry) async {
        guard let url = URL(string: "\(bridgeBaseURL)/modules/caffeineTracker/data?id=\(entry.id)") else { return }

        do {
            var request = URLRequest(url: url)
            request.httpMethod = "DELETE"
            let (_, _) = try await URLSession.shared.data(for: request)
            entries.removeAll { $0.id == entry.id }
            cacheData()
        } catch {
            errorMessage = error.localizedDescription
        }
    }

    // MARK: - Quick Add

    func quickAdd(drinkType: DrinkType) async {
        let caffeine = Self.defaultCaffeineMg(drinkType: drinkType, size: .medium)
        let entry = CaffeineTrackerEntry(
            drinkType: drinkType,
            drinkSize: .medium,
            caffeineAmount: caffeine,
            loggedAt: Date()
        )
        await addEntry(entry)
    }

    // MARK: - Cache

    private func cacheData() {
        let encoder = JSONEncoder()
        encoder.dateEncodingStrategy = .iso8601
        if let data = try? encoder.encode(entries) {
            UserDefaults.standard.set(data, forKey: "dynamic_module_caffeineTracker_cache")
        }
    }

    // MARK: - Date Helpers

    private var calendar: Calendar { Calendar.current }

    private func isToday(_ date: Date) -> Bool {
        calendar.isDateInToday(date)
    }

    private func startOfDay(_ date: Date) -> Date {
        calendar.startOfDay(for: date)
    }

    private func entriesForDay(_ date: Date) -> [CaffeineTrackerEntry] {
        entries.filter { calendar.isDate($0.loggedAt, inSameDayAs: date) }
    }

    private func totalMgForDay(_ date: Date) -> Int {
        entriesForDay(date).reduce(0) { $0 + $1.caffeineAmount }
    }

    // MARK: - Computed Properties: Today

    var currentCaffeineInSystem: Double {
        let now = decayTick
        return entries.reduce(0.0) { total, entry in
            let elapsedHours = now.timeIntervalSince(entry.loggedAt) / 3600.0
            guard elapsedHours >= 0 else { return total }
            return total + Double(entry.caffeineAmount) * pow(0.5, elapsedHours / 5.0)
        }
    }

    var todayTotalMg: Int {
        entries.filter { isToday($0.loggedAt) }
            .reduce(0) { $0 + $1.caffeineAmount }
    }

    var todayDrinkCount: Int {
        entries.filter { isToday($0.loggedAt) }.count
    }

    var dailyLimitProgress: Double {
        Double(todayTotalMg) / Double(dailyGoal)
    }

    var isOverLimit: Bool {
        todayTotalMg > dailyGoal
    }

    var safeToSleepTime: Date? {
        let now = decayTick
        let threshold = 50.0
        guard currentCaffeineInSystem > threshold else { return nil }

        // Binary search for the hour offset when caffeine drops below threshold
        var low = 0.0
        var high = 48.0

        for _ in 0..<50 {
            let mid = (low + high) / 2.0
            let futureTime = now.addingTimeInterval(mid * 3600)
            let caffeineAtTime = entries.reduce(0.0) { total, entry in
                let elapsed = futureTime.timeIntervalSince(entry.loggedAt) / 3600.0
                guard elapsed >= 0 else { return total }
                return total + Double(entry.caffeineAmount) * pow(0.5, elapsed / 5.0)
            }
            if caffeineAtTime > threshold {
                low = mid
            } else {
                high = mid
            }
        }

        return now.addingTimeInterval(high * 3600)
    }

    var lastCaffeineTime: Date? {
        entries.filter { isToday($0.loggedAt) }
            .max(by: { $0.loggedAt < $1.loggedAt })?
            .loggedAt
    }

    var todayDrinks: [CaffeineTrackerEntry] {
        entries.filter { isToday($0.loggedAt) }
            .sorted { $0.loggedAt < $1.loggedAt }
    }

    // MARK: - Computed Properties: Trends

    var weeklyDailyTotals: [(Date, Int)] {
        let today = startOfDay(Date())
        return (0..<7).reversed().map { daysAgo in
            let date = calendar.date(byAdding: .day, value: -daysAgo, to: today)!
            return (date, totalMgForDay(date))
        }
    }

    var weeklyAverage: Double {
        let totals = weeklyDailyTotals.map(\.1)
        guard !totals.isEmpty else { return 0 }
        return Double(totals.reduce(0, +)) / Double(totals.count)
    }

    var previousWeekAverage: Double {
        let today = startOfDay(Date())
        let totals = (7..<14).map { daysAgo -> Int in
            let date = calendar.date(byAdding: .day, value: -daysAgo, to: today)!
            return totalMgForDay(date)
        }
        guard !totals.isEmpty else { return 0 }
        return Double(totals.reduce(0, +)) / Double(totals.count)
    }

    var weekOverWeekChange: Double {
        guard previousWeekAverage > 0 else { return 0 }
        return ((weeklyAverage - previousWeekAverage) / previousWeekAverage) * 100.0
    }

    var streakDaysUnderLimit: Int {
        let today = startOfDay(Date())
        var streak = 0
        let startOffset = isOverLimit ? 1 : 0

        for daysAgo in startOffset..<365 {
            guard let date = calendar.date(byAdding: .day, value: -daysAgo, to: today) else { break }
            if totalMgForDay(date) <= dailyGoal {
                streak += 1
            } else {
                break
            }
        }

        return streak
    }

    var averageLastCaffeineHour: Double? {
        let today = startOfDay(Date())
        var lastHours: [Double] = []

        for daysAgo in 0..<7 {
            guard let date = calendar.date(byAdding: .day, value: -daysAgo, to: today) else { continue }
            let dayEntries = entriesForDay(date)
            if let lastEntry = dayEntries.max(by: { $0.loggedAt < $1.loggedAt }) {
                let components = calendar.dateComponents([.hour, .minute], from: lastEntry.loggedAt)
                if let hour = components.hour, let minute = components.minute {
                    lastHours.append(Double(hour) + Double(minute) / 60.0)
                }
            }
        }

        guard lastHours.count >= 2 else { return nil }
        return lastHours.reduce(0, +) / Double(lastHours.count)
    }

    var monthlyIntensityMap: [Date: Double] {
        let today = Date()
        guard let range = calendar.range(of: .day, in: .month, for: today),
              let monthStart = calendar.date(from: calendar.dateComponents([.year, .month], from: today)) else {
            return [:]
        }

        var map: [Date: Double] = [:]
        for day in range {
            guard let date = calendar.date(byAdding: .day, value: day - 1, to: monthStart) else { continue }
            let dayStart = startOfDay(date)
            let total = totalMgForDay(dayStart)
            let intensity = min(Double(total) / Double(dailyGoal), 1.0)
            map[dayStart] = intensity
        }
        return map
    }

    // MARK: - Chart Data

    var weeklyChartData: [ChartDataPoint] {
        let formatter = DateFormatter()
        formatter.dateFormat = "EEE"
        return weeklyDailyTotals.map { date, total in
            ChartDataPoint(label: formatter.string(from: date), value: Double(total))
        }
    }

    // MARK: - Insights

    var insights: [ModuleInsight] {
        var result: [ModuleInsight] = []

        if isOverLimit {
            result.append(ModuleInsight(
                type: .warning,
                title: "Over Daily Limit",
                message: "You've consumed \(todayTotalMg)mg today, exceeding the \(dailyGoal)mg limit."
            ))
        }

        if previousWeekAverage > 0 {
            let change = weekOverWeekChange
            if change > 10 {
                result.append(ModuleInsight(
                    type: .trend,
                    title: "Intake Increasing",
                    message: String(format: "Weekly average is up %.0f%% compared to last week.", change)
                ))
            } else if change < -10 {
                result.append(ModuleInsight(
                    type: .trend,
                    title: "Intake Decreasing",
                    message: String(format: "Weekly average is down %.0f%% from last week. Nice!", abs(change))
                ))
            }
        }

        if streakDaysUnderLimit >= 7 {
            result.append(ModuleInsight(
                type: .achievement,
                title: "\(streakDaysUnderLimit)-Day Streak!",
                message: "You've stayed under the daily limit for \(streakDaysUnderLimit) consecutive days."
            ))
        } else if streakDaysUnderLimit >= 3 {
            result.append(ModuleInsight(
                type: .achievement,
                title: "\(streakDaysUnderLimit)-Day Streak",
                message: "Keep going! \(7 - streakDaysUnderLimit) more days for a full week."
            ))
        }

        if let avgHour = averageLastCaffeineHour, avgHour >= 16 {
            let hour = Int(avgHour)
            let minute = Int((avgHour - Double(hour)) * 60)
            result.append(ModuleInsight(
                type: .suggestion,
                title: "Late Caffeine Intake",
                message: String(format: "Average last caffeine at %d:%02d. Try cutting off by 2 PM for better sleep.", hour, minute)
            ))
        }

        if let sleepTime = safeToSleepTime {
            let formatter = DateFormatter()
            formatter.dateFormat = "h:mm a"
            result.append(ModuleInsight(
                type: .suggestion,
                title: "Safe to Sleep",
                message: "Caffeine drops below 50mg around \(formatter.string(from: sleepTime))."
            ))
        }

        return result
    }
}

struct CaffeineTrackerEntry: Codable, Identifiable {
    var id: String = UUID().uuidString
    var date: String = {
        let f = DateFormatter()
        f.dateFormat = "yyyy-MM-dd HH:mm"
        return f.string(from: Date())
    }()
    var drinkType: DrinkType
    var drinkSize: DrinkSize
    var caffeineAmount: Int
    var loggedAt: Date
    var notes: String?

    var summaryLine: String {

        parts.append("\(drinkSize)")
        parts.append("\(caffeineAmount)")
        parts.append("\(loggedAt)")
        if let v = notes { parts.append("\(v)") }
        return parts.joined(separator: " | ")
    }
}
